                                                 TO RUN AND STOP A CONTAINER
 
sudo docker run -d -it --name 'new cont name' -p 80:80 'image' (TO RUN A CONT)
sudo docker run -d -it --name 'new cont name' -p 'image' (TO CHOOSE A RANDOM PORT)
sudo docker run -d -it --name 'new cont name' -p 81:80 'image' /bin/bash (TO RUN A OS IMAGES)
sudo docker run -d -it --name 'new cont name' -p 80:80 --mount source='volume name',destination=/data 'image' (TO ADD VOLUME IN CONT)
sudo docker run -d -it --name 'new cont name' -p 80:80 -v /data:'destination new name' 'image' (TO BIND THE VOLUME)
sudo docker ps (TO DIPLAY A PROCESSING CONT)
sudo docker exec -it 'cont name' /bin/sh (TO EXECUTE THE CONTAINER)
sudo docker stop 'cont name' (TO STOP THE CONT)
sudo docker start 'cont name' (TO START THE CONT)
sudo docker rm 'cont name' (to remove cont)   sudo docker rm -f 'cont name' (WARNING TO REMOVE FORCE WORKING CONT)
sudo docker rmi 'image' (TO REMOVE IMAGES)
sudo docker logs 'cont name' (TO DISPLAY LOGS OF THE CONT)
sudo docker top 'cont name' (TO DISPLAY CONT PROCESS)
sudo docker inspect 'image:tag' (TO DISPLAY THE OUTPUT)
sudo docker history 'image:tag' (TO DIPLAY THE HISTORY)
sudo docker save 'image id' > newfile.tar (TO BACKUP THE IMAGES)
scp file.tar 'inet ip'://home/pc/ (TO MOVE ONE HOST TO ANOTHER HOST)
sudo docker load -i 'file.tar' (TO EXRACT THE BACKUP FILE)
sudo docker commit 'cont id' 'new image name:tag' (AFTER CHANGES IN CONT)

                                       Instruction & arguments in Docker file

Instruction	Description
FROM
ADD	        Add local or remote files and directories.
ARG	        Use build-time variables.
CMD	        Specify default commands.
COPY	        Copy files and directories.
ENTRYPOINT	Specify default executable.
ENV	        Set environment variables.
EXPOSE	        Describe which ports your application is listening on.
FROM	        Create a new build stage from a base image.
HEALTHCHECK	Check a container's health on startup.
LABEL	        Add metadata to an image.
MAINTAINER	Specify the author of an image.
ONBUILD	        Specify instructions for when the image is used in a build.
RUN	        Execute build commands.
SHELL	        Set the default shell of an image.
STOPSIGNAL	Specify the system call signal for exiting a container.
USER	        Set user and group ID.
VOLUME	        Create volume mounts.
WORKDIR	        Change working DIRECTORY

                                              TO CREATE SWARM
sudo docker node ls (TO LIST ALL SWARM)
sudo docker swarm init --advertise-addr 'net ip add'(TO ADD NODE)
sudo docker service ls (TO LIST ALL SERVICE)
sudo docker service ps 'cont name' (TO DISPLAY HOW MANY NODES ARE CONNECTED)
sudo docker network ls (TO SEE DRIVERS)
sudo docker network create -d overlay 'newoverlay' (TO CREATE A OVERLAY DRIVER)
sudo docker service rm 'cont name' (TO STOP THE SERVICE)

sudo docker node promote 'hostname' (TO PROMOTE THE HOST)
sudo docker node demote 'hostname' (TO DEMOTE THE HOST)

sudo docker service create --replicas 2 -p 80:80 --name 'cont' 'image' (TO CREATE 2 NODES )
sudo docker service create --mode global -p 80:80 --name 'cont' 'image' (TO CREATE ALL NODES CONT) 

sudo docker service scale 'cont=5' (TO SCALE UP THE CONTAINER)
sudo docker service scale 'cont=2' (TO SCALE DOWN THE CONTAINER) 

sudo docker service create --replicas 2 -p 80:80 --name 'cont' 'image:1.1' (TO ROLLBACK OLDER VERSION)
sudo docker service update --image 'image:latest' 'cont name' (HIGHER VERSION ROLLING UPDATE)
sudo docker service rollback 'cont name' (TO UPDATE PREVIOUS VERSION)

sudo docker node update --availability drain 'hostname' (TO EMPTY THE HOST)
sudo docker node update --availability pause 'hostname' (TO PAUSE THE HOST) 
sudo docker node update --availability active 'hostname' (TO ACTIVATE THE HOST)
